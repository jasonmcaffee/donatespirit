CREATE TABLE donatespirit.users
(
    id int AUTO_INCREMENT PRIMARY KEY NOT NULL,
    username char(50) NOT NULL,
    password char(50) NOT NULL
);
ALTER TABLE donatespirit.users ADD CONSTRAINT unique_username UNIQUE (username);

CREATE TABLE `UserInfo` (
  `id` int(11) AUTO_INCREMENT NOT NULL,
  `userId` int(11) NOT NULL,
  `email` varchar(100) DEFAULT NULL,
  `creationIp` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `unique_userId` (`userId`),
  CONSTRAINT `userinfo_ibfk_1` FOREIGN KEY (`userId`) REFERENCES `User` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1


CREATE TABLE donatespirit.Message
(
    id int PRIMARY KEY AUTO_INCREMENT NOT NULL,
    userId int NOT NULL,
    message varchar(5000) NOT NULL,
    date datetime NOT NULL,
    ip varchar(25),
    CONSTRAINT `user_ibfk_1` FOREIGN KEY (`userId`) REFERENCES `User` (`id`)
);
